{{- with .Params.series -}} {{- $seriesSlug := index . 0 -}} {{- $currentPart := $.Params.part -}} {{- $allPosts := where $.Site.RegularPages "Params.series" "intersect" (slice $seriesSlug) -}} {{- $sorted := sort $allPosts "Params.part" -}} {{- $prevPosts := where $sorted "Params.part" "lt" $currentPart -}} {{- $nextPosts := where $sorted "Params.part" "gt" $currentPart -}} {{- $prevPost := index (last 1 $prevPosts) 0 -}} {{- $nextPost := index (first 1 $nextPosts) 0 -}} {{- if or $prevPost $nextPost -}}
{{- with $prevPost }} « {{ i18n "prev_page" }} {{- .Name -}} {{- end }} {{- with $nextPost }} {{ i18n "next_page" }} » {{- .Name -}} {{- end }}
{{- end -}} {{- else -}} {{- $pages := where site.RegularPages "Type" "in" site.Params.mainSections }} {{- if and (gt (len $pages) 1) (in $pages . ) }}
{{- with $pages.Next . }} « {{ i18n "prev_page" }} {{- .Name -}} {{- end }} {{- with $pages.Prev . }} {{ i18n "next_page" }} » {{- .Name -}} {{- end }}
{{- end -}} {{- end -}}
